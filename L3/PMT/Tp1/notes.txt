creer un alias :
		alias gcc = 'gcc -Wall -ansi -pedantic -Werror'
pour tenir compte des modifications
		source (.tcshrc | .bashrc)

  unzip -l tuto.zip
pour decompresser
  unzip tuto.zip
compresser
  zip -r tuto.zip tuto/
